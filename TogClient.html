Gets a session, or creates a new one if it doesn't exist
Example
tog.session('default', 'abc123', 60*60*24) // expires in 1 day
  .then(session => {
    console.log('Experiment: ' + session.experiment)
    console.log('Flags: ' + JSON.stringify(session.flags))
  })
